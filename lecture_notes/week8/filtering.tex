\subsection*{The filtering problem}
Let
\begin{align}
dX_t &= AX_tdt + dW_t\\
Y_t &= c^TX_t
\end{align}
Typically we may be interested in the mean, hence $c = 1/n$. 
We assume $X_t$ be stationary. 
Let ${\bf Y}_{l,t}= (Y_t,Y_{t-\delta},Y_{t-2\delta},\dots,Y_{t-l\delta})^T$
and our coal is to forecast: 
\begin{equation}
{\mathbb E}[f(Y_{t+s})|{\bf Y}_{l,t}]
\end{equation} 
where the expectation is taken over both the trajectory of $X_t$ and the prior distribution of $A$. 

 The conditional expectations
\begin{equation}
{\mathbb E}[f(Y_{t+s})|{\bf Y}_{l,t},A]
\end{equation}
are straightforward to compute since $Y_{t+s}|{\bf Y}_{l,t}$ is Gaussian which can be obtained as follows. 
 The stationary covariance matrix of $X_t$ satisfies the Lyaponuv equation 
\begin{equation}
A\Pi + \Pi A^T = - I
\end{equation}
Then $Y_t$ is a mean zero Gaussian process with kernel $K(s,t)$ given by 
\begin{equation}
K_y(t,s) = {\rm cov}(Y_t,Y_s) 
=
\begin{cases}
c^T \Pi e^{A^T (t-s)}c, & t\ge s,\\[4pt]
c^T e^{A(s-t)}\Pi c, & s>t,
\end{cases}
\end{equation} 


 Specifically,  
\begin{equation}
Y_{t+s}|{\bf Y}_t^l,A \sim {\rm Normal}({\bf k}_{l,s}^T {\bf Y}_{l,t},v_{l,s}). 
\end{equation}
where 
\begin{align}
\Sigma_l(i,j) &:= K_y(t-i\delta,\;t-j\delta),\qquad 0\le i,j\le l,\\
\kappa_{l,s}(i) &:= K_y(t+s,\;t-i\delta),\qquad 0\le i\le l,\\
\sigma_s^2 &:= K_y(t+s,t+s),\\[4pt]
{\bf k}_{l,s} &= \kappa_{l,s}^T \Sigma_l^{-1},\\
v_{l,s} &= \sigma_s^2 - \kappa_{l,s}^T\Sigma_l^{-1}\kappa_{l,s}.
\end{align}
We will use $\rho_{t+s}(y|{\bf Y}_t^l,A,B)$ to denote the corresponding density.
\begin{equation}
\rho_{t+s}(y\mid \mathbf Y_t^{(l)})
=\iint \rho_{t+s}(y\mid \mathbf Y_t^{(l)},A,B)\,p(A,B\mid \mathbf Y_t^{(l)})\,dA\,dB.
\end{equation}
